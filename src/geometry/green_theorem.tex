\subsection{Green's Theorem}

반시계방향의 조각적으로 매끄러운 단순폐곡선 $C$와 $C$로 유계된 영역 $D$에 대해
\[
\iint_D \left( \frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y} \right)dA
=
\int_C P\,dx+Q\,dy
\]

다각형의 각 변에 대한 합으로 계산하면
\[
\int_C P\,dx+Q\,dy
=
\sum_{i=1}^{N}
\int_{(x_i,y_i)}^{(x_{i+1},y_{i+1})}
P\,dx+Q\,dy
\]

\vspace{5pt}
\noindent \textbf{Examples} \\

\noindent \textbf{1. Polar Moment} \\
\[
P=-\frac{1}{3}y^3,\qquad Q=\frac{1}{3}x^3
\]
\[
\iint_D (x^2+y^2)\,dA
=
\oint_C \frac{1}{3}x^3\,dy-\frac{1}{3}y^3\,dx
\]
\[
=
\sum_{i=1}^{N}
\int_{(x_i,y_i)}^{(x_{i+1},y_{i+1})}
\frac{1}{3}x^3\,dy-\frac{1}{3}y^3\,dx
\]

각 변에서
\[
x=x_i+(x_{i+1}-x_i)t,\qquad
y=y_i+(y_{i+1}-y_i)t,\qquad 0\le t\le 1
\]
\[
dx=(x_{i+1}-x_i)dt,\qquad
dy=(y_{i+1}-y_i)dt
\]

\[
\resizebox{\linewidth}{!}{$\displaystyle
=
\frac{1}{3}
\sum_{i=1}^{N}
\int_0^1
\left(
(x_i+(x_{i+1}-x_i)t)^3(y_{i+1}-y_i)
-
(y_i+(y_{i+1}-y_i)t)^3(x_{i+1}-x_i)
\right)dt
$}
\]

\[
\resizebox{\linewidth}{!}{$\displaystyle
=
\frac{1}{12}
\sum_{i=1}^{N}
(x_i y_{i+1}-x_{i+1}y_i)
\left(
x_i^2+x_i x_{i+1}+x_{i+1}^2
+
y_i^2+y_i y_{i+1}+y_{i+1}^2
\right)
$}
\]

\vspace{5pt}
\noindent \textbf{2. First Moment} \\
\[
\resizebox{\linewidth}{!}{$\displaystyle
\iint_D x\,dA
=
\oint_C \frac{1}{2}x^2\,dy
=
\frac{1}{6}
\sum_{i=1}^{N}
(y_{i+1}-y_i)
(x_i^2+x_i x_{i+1}+x_{i+1}^2)
$}
\]

\vspace{5pt}
\noindent \textbf{Area} \\
\[
S=\iint_D 1\,dA
\]

\[
P=-\frac{y}{2},\qquad Q=\frac{x}{2},
\qquad
\left(
\frac{\partial Q}{\partial x}
-
\frac{\partial P}{\partial y}
\right)=1
\]
\[
\therefore
S
=
\int_C (P\,dx+Q\,dy)
=
\frac{1}{2}\oint_C (x\,dy-y\,dx)
\]

\vspace{5pt}
\noindent \textbf{Circular Arc Integral} \\
\[
x=x_c+r\cos\theta,\qquad y=y_c+r\sin\theta
\]

\[
\resizebox{\linewidth}{!}{$\displaystyle
S
=
\frac{1}{2}\oint (x\,dy-y\,dx)
=
\frac{r}{2}
\left[
x_c\sin\theta-y_c\cos\theta+r\theta
\right]_{\theta_1}^{\theta_2}
$}
\]

\vspace{5pt}
\noindent \textbf{Line Segment Integral} \\
\[
v_x=x_2-x_1,\qquad v_y=y_2-y_1
\]

\[
x=x_1+v_x t,\qquad y=y_1+v_y t,\qquad 0\le t\le 1
\]

\[
S
=
\frac{1}{2}\oint (x\,dy-y\,dx)
=
\frac{1}{2}
\left[
(v_yx_1-v_xy_1)t
\right]_0^1
\]

\[
\resizebox{\linewidth}{!}{$\displaystyle
=
\frac{1}{2}
(y_2x_1-y_1x_1-x_2y_1+x_1y_1)
=
\frac{1}{2}(x_1y_2-x_2y_1)
$}
\]
